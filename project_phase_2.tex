\section{Introduction}
\begin{frame}{Problem Statement \& Need of the System}
    \vspace{-2mm}
    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{\small Problem Statement}
        \small 
        To develop a privacy-first, edge-based AI speech therapy system for children with Autism Spectrum Disorder (ASD) in rural South India. Current cloud-based tools struggle with Dravidian retroflex consonants and atypical ASD prosody while exposing pediatric privacy risks. The offline system addresses therapist scarcity with syllable-level feedback and on-device tracking.
    \end{block}

    \vspace{2mm}

    \setbeamercolor{block title}{bg=black, fg=white}
    \begin{block}{\small Need of the System}
        \footnotesize
        \begin{itemize}
            \setlength{\itemsep}{4pt}
            \item \textbf{Addressing Therapist Scarcity:} With fewer than one therapist per 10,000 children in rural Tamil Nadu, one remote therapist can monitor 20--30 children.
            \item \textbf{Clinical Precision (Syllable-Level):} Replaces binary whole-word scoring with syllable-boundary evaluation.
            \item \textbf{Privacy \& Data Sovereignty:} Processing and deleting raw audio on-device keeps pediatric voice data local.
            \item \textbf{Objective Progress Tracking:} Replaces subjective observation with VSA and GOP metrics.
        \end{itemize}
    \end{block}
\end{frame}
\begin{frame}{Project Objectives}
    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{Core Technical \& Clinical Objectives}
        \small
        \begin{itemize}
            \setlength{\itemsep}{6pt}
            \item \textbf{Edge Computing Pipeline:} Develop an offline AI pipeline that processes speech on-device.
            \item \textbf{Transfer Learning for Dataset Scarcity:} Adapt Wav2Vec 2.0 through two-stage transfer learning for Tamil phonetics and pediatric disordered speech.
            \item \textbf{Acoustic Space Normalization:} Implement VTLN so children are not penalized for physiology.
            \item \textbf{Syllable-Level GOP Scoring:} Use GOP to identify mispronounced syllables.
            \item \textbf{Longitudinal Metric Tracking:} Track VSA and GOP trends per child, and deliver reports.
            \item \textbf{Adaptive Scaffolding:} Build a staged difficulty engine that transitions to DDPG.
        \end{itemize}
    \end{block}
\end{frame}

\section{Novelty}
\begin{frame}{Our Novelty \& Architectural Innovations (1/2)}
    \vspace{-2mm}
    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{\small 1. Syllable-Level GOP Scoring for Regional Languages}
        \footnotesize
        Existing AI therapy systems score pronunciation at the word level, producing pass/fail data. We use Dynamic Time Warping (DTW) to align child audio at syllable boundaries and identify failed articulation.
    \end{block}
    \vspace{2mm}
    
    \setbeamercolor{block title}{bg=black, fg=white}
    \begin{block}{\small 2. Two-Stage Transfer Learning}
        \footnotesize
        Because no Tamil ASD speech corpus exists, training from scratch is not feasible. We fine-tune Wav2Vec 2.0 in two stages: first on AI4Bharat Kathbath for Dravidian phonetics, then on TORGO for disordered speech.
    \end{block}
    \vspace{2mm}

    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{\small 3. VTLN Child-to-Adult Acoustic Mapping}
        \footnotesize
        Children's shorter vocal tracts (10-12cm) produce higher formant frequencies than adult references (17cm). We apply VTLN before inference so the model measures pronunciation quality.
    \end{block}
\end{frame}

% ============================================================
\begin{frame}{Our Novelty \& Architectural Innovations (2/2)}
    \vspace{-2mm}
    \setbeamercolor{block title}{bg=black, fg=white}
    \begin{block}{\small 4. Staged Adaptive Difficulty (Rule-Based to Federated DDPG)}
        \footnotesize
        We propose a staged approach. Phase 1 deploys an auditable rule-based engine that manages therapy difficulty while aggregating interaction data. Phase 2 transitions to a DDPG agent after sufficient data supports stable policy decisions.
    \end{block}
    \vspace{2mm}
    
    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{\small 5. Privacy-First Federated Learning Without FHE Overhead}
        \footnotesize
        Rigorous privacy on low-cost tablets rules out computationally heavy Fully Homomorphic Encryption. Our system achieves zero-trust privacy: raw audio is processed and deleted on-device, and only model weight gradients are synced.
    \end{block}
    \vspace{2mm}

    \setbeamercolor{block title}{bg=black, fg=white}
    \begin{block}{\small 6. Automated Bi-Weekly Therapist Dashboards}
        \footnotesize
        Using a MongoDB aggregation pipeline, the system queries session documents to detect recurring phoneme weaknesses. It then generates a report that alerts a therapist when specific consonant or vowel classes need intervention.
    \end{block}
\end{frame}

\section{Methodology}
\begin{frame}{Technical Knowledge: Core Algorithms in Action}
    \vspace{-2mm}
    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{\small Wav2Vec 2.0 \& Dynamic Time Warping (DTW)}
        \footnotesize
        \textbf{How it works for our project:} Wav2Vec 2.0 uses convolutional layers to extract acoustic features, such as the burst of retroflex consonants, and a Transformer to capture long-range prosody. DTW aligns irregular ASD speech with a healthy reference template, allowing syllable boundaries to be isolated regardless of speaking speed.
    \end{block}
    \vspace{1mm}
    
    \setbeamercolor{block title}{bg=black, fg=white}
    \begin{block}{\small Goodness of Pronunciation (GOP)}
        \footnotesize
        \textbf{How it works for our project:} Instead of outputting text, GOP computes a confidence score for each isolated syllable from the acoustic model's likelihoods. Scores below the threshold flag articulation failures and provide actionable feedback.
    \end{block}
    \vspace{1mm}
    
    \setbeamercolor{block title}{bg=amritaMaroon, fg=white}
    \begin{block}{\small Vowel Space Area (VSA) \& DDPG Scaffolding}
        \footnotesize
        \textbf{How it works for our project:} VSA tracks the frequencies of core corner vowels across sessions. An expanding triangle area indicates better jaw and tongue motor control. The VSA trend, combined with real-time GOP scores, forms the environmental "state" used by the DDPG agent to fine-tune the next therapy prompt.
    \end{block}
\end{frame}

\section{Challenges}
\begin{frame}{Technical \& Research Challenges}
    \vspace{-3mm}
    \setbeamercolor{cbox}{bg=amritaMaroon!10, fg=black}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            
            \begin{beamercolorbox}[wd=\linewidth, sep=3pt, rounded=true]{cbox}
                \textbf{\textcolor{amritaMaroon}{\footnotesize 1. Complete Dataset Scarcity}} \\[1mm]
                {\scriptsize \textbf{Challenge:} Zero public samples of Tamil pediatric ASD speech.\\
                \textbf{Solution:} Two-stage transfer learning (AI4Bharat + TORGO proxy) \& primary clinical data collection.\par}
            \end{beamercolorbox}
            \vspace{2mm}
            
            \begin{beamercolorbox}[wd=\linewidth, sep=3pt, rounded=true]{cbox}
                \textbf{\textcolor{amritaMaroon}{\footnotesize 2. Child-Adult Acoustic Mismatch}} \\[1mm]
                {\scriptsize \textbf{Challenge:} Adult models falsely penalize shorter child vocal tracts (higher formants).\\
                \textbf{Solution:} VTLN mapping normalizes audio so scores reflect articulation effort.\par}
            \end{beamercolorbox}
            \vspace{2mm}
            
            \begin{beamercolorbox}[wd=\linewidth, sep=3pt, rounded=true]{cbox}
                \textbf{\textcolor{amritaMaroon}{\footnotesize 3. Irregular Syllable Boundaries}} \\[1mm]
                {\scriptsize \textbf{Challenge:} ASD speech contains unexpected pauses and heavily elongated phonemes.\\
                \textbf{Solution:} DTW cost-matrix alignment optimizes boundaries without fixed durations.\par}
            \end{beamercolorbox}
            
        \end{column}

        \begin{column}{0.48\textwidth}
            
            \begin{beamercolorbox}[wd=\linewidth, sep=3pt, rounded=true]{cbox}
                \textbf{\textcolor{amritaMaroon}{\footnotesize 4. Edge Latency Budgets}} \\[1mm]
                {\scriptsize \textbf{Challenge:} Running Wav2Vec 2.0 under 200ms on low-cost Android tablets.\\
                \textbf{Solution:} TinyML 8-bit quantization and student-teacher distillation compress the model.\par}
            \end{beamercolorbox}
            \vspace{2mm}
            
            \begin{beamercolorbox}[wd=\linewidth, sep=3pt, rounded=true]{cbox}
                \textbf{\textcolor{amritaMaroon}{\footnotesize 5. DDPG Data Convergence Limits}} \\[1mm]
                {\scriptsize \textbf{Challenge:} Deep RL needs thousands of episodes; day-1 deployment is clinically unsafe.\\
                \textbf{Solution:} Phase 1 rule-based engine harvests FL data until the DDPG threshold is met.\par}
            \end{beamercolorbox}
            \vspace{2mm}
            
            \begin{beamercolorbox}[wd=\linewidth, sep=3pt, rounded=true]{cbox}
                \textbf{\textcolor{amritaMaroon}{\footnotesize 6. Offline Reliability Constraints}} \\[1mm]
                {\scriptsize \textbf{Challenge:} Unstable internet connectivity in rural homes disrupts therapy.\\
                \textbf{Solution:} Fully local SQLite processing architecture with opportunistic MongoDB Atlas syncs.\par}
            \end{beamercolorbox}
            
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Conclusion}
    \vspace{2mm}
    \setbeamercolor{block title}{bg=black, fg=white}
    \begin{block}{A Targeted Solution for a Critical Gap}
        \small
        This project is not a generic speech-to-text wrapper. Each technical decision addresses a specific clinical or geographic gap. The system is designed to provide therapist-level precision within the 3-to-8 year early intervention window, in settings where internet access and specialized clinics are unavailable.
    \end{block}
    
    \vspace{2mm}
    \textbf{Summary of Technical Justifications:}
    \begin{itemize}
        \item \textbf{VTLN} addresses the acoustic differences between children and adults.
        \item \textbf{Two-Stage Transfer Learning} bypasses the absence of Tamil ASD corpora.
        \item \textbf{Syllable-Level DTW} replaces clinically unactionable whole-word grading.
        \item \textbf{Federated Edge AI} is required because pediatric biometric privacy is legally non-negotiable.
    \end{itemize}

    \vspace{2mm}
    \begin{beamercolorbox}[sep=4pt, center, rounded=true]{palette primary}
